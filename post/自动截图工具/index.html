需求：#

视频的知识密度太高，一时间难以理解，最好的办法就是保留好这个视频的字幕截图，在自动拼接好。

实现思路：#

  • 鼠标事件获取鼠标按下和释放的位置
  • 注册快捷键，每按一次就对固定区域截图
  • 图片拼接

实现#

  • 鼠标事件获取鼠标按下和释放的位置

    本来是被这个join事件困惑住，原来直接在子事件里stop就好了。

    # Collect events until released
    with mouse.Listener(
            # on_move=on_move,

  • 注册快捷键，每按一次就对固定区域截图

    没有把单张图片保存，直接读到内存里。f2为截图的快捷键，f4为停止截图的快捷键

def screen_it():
    global file
    img = ImageGrab.grab(bbox=(press_pos[0], press_pos[1],
                               release_pos[0], release_pos[1]))
    file.append(img)
    print("已经截图")
file=[]
keyboard.add_hotkey('f2',screen_it)
keyboard.wait('f4')

  • 图片拼接

    拼接就是用PIL中的拼接，文件保存后调用os.startfile()就可以用系统默认应用程序打开文件了~

# -*- coding:utf-8 -*-
# __author__ = "shiluanzzz"
from pynput import mouse
import keyboard
import pyautogui
import time
from PIL import ImageGrab,Image
import os

def on_move(x, y):
    print('Pointer moved to {0}'.format(
        (x, y)))


def on_click(x, y, button, pressed):
    global press_pos, flag
    global release_pos
    print('{0} at {1}'.format('Pressed' if pressed else 'Released',
                              (x, y)))
    if pressed:
        press_pos = (x, y)
    else:
        release_pos = (x, y)
        if release_pos[0]-press_pos[0]>30:
            listener.stop()



def on_scroll(x, y, dx, dy):
    print('Scrolled {0} at {1}'.format(
        'down' if dy < 0 else 'up',
        (x, y)))

def screen_it():
    global file
    img = ImageGrab.grab(bbox=(press_pos[0], press_pos[1], release_pos[0], release_pos[1]))
    file.append(img)
    print("已经截图")

def concat_img(file):

    width,height=file[0].size
    result = Image.new(file[0].mode,(width,height*len(file)))
    for i,im in enumerate(file):
        result.paste(im,box=(0,i*height))
    file_name='result{}.jpg'.format(time.strftime('%H-%M-%S', time.localtime()))
    result.save(file_name)
    return file_name
if __name__ == '__main__':
    flag = False
    press_pos = ()
    release_pos = ()
    file=[]

    # Collect events until released
    with mouse.Listener(
            # on_move=on_move,
            on_click=on_click,
            on_scroll=on_scroll) as listener:
        listener.join()
    print("截图位置获取成功")
    keyboard.add_hotkey('f2',screen_it)
    keyboard.wait('f4')
    fp=concat_img(file)
    print("图片保存成功")
    os.startfile(fp)
